from collections import defaultdict
def solution(gems):
    answer = [0,100000]
    gemSet = list(set(gems))
    gemCnt = len(gemSet)
    isCnt = defaultdict(int)
    lp = 0
    rp = 0
    isCnt[gems[0]] += 1
    while 1:
        if len(isCnt) != gemCnt:
            if rp == len(gems)-1:
                break
            rp += 1
            isCnt[gems[rp]] += 1
        elif len(isCnt) == gemCnt:
            if answer[1]-answer[0] > rp-lp: answer = [lp+1,rp+1]
            if (rp-lp == gemCnt-1):
                break
            if isCnt[gems[lp]] > 1: isCnt[gems[lp]] -= 1
            else: del isCnt[gems[lp]]
            lp += 1
    return answer
