## 백준 - 1874[스택 수열]

#### <코드>  

<pre>
<code>
import sys

n = int(sys.stdin.readline())

l = []
for i in range(n):
    l.append(int(sys.stdin.readline()))
stack = []
idx = 0
cur = 1
answer = []
while (cur <= n):
    stack.append(cur)
    answer.append('+')
    cur += 1
    while 1:
        if len(stack) != 0 and stack[-1] == l[idx]:
            idx += 1
            stack.pop()
            answer.append('-')
        else: break
if idx == n:
    for i in answer:
        print(i)
else: print("NO")
