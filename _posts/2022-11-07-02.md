from collections import defaultdict
def solution(id_list, report, k):
    answer = []
    reportHash = defaultdict(set)
    stoped = defaultdict(int)
    for s in report:
        a,b = s.split(" ")
        l = len(reportHash[a])
        reportHash[a].add(b)
        if l != len(reportHash[a]):
            stoped[b] += 1
    for i in id_list:
        sum = 0
        for j in reportHash[i]:
            if stoped[j] >= k:
                sum += 1
        answer.append(sum)
    return answer
